es6新语法

Veröffentlicht am 2018-03-29

块级作用域

ES5里，JS中作用域有：全局作用域、函数作用域，没有块作用域的概念。ECMAScript 6(简称ES6)中新增了块级作用域。

块级作用域由 {} 包括，if语句和for语句里面的 {} 也属于块作用域。（严格模式下 function 变量属于块级作用域）

例如 {
var a = 1;
}
console.log(a); // 输出1，可见通过var定义的变量可以跨块作用域访问到。

(function A() {
    var b = 2;
    console.log(b); // 输出2
})();
console.log(b); // 报错，可见通过var定义的变量不能跨函数作用域访问到

if(true) {
    var c = 3;
}
console.log(c); // 3
for(var i = 0; i < 4; i ++) {
    var d = 5;
};
console.log(i); // 输出4   (循环结束i已经是4，所以此处i为4)
console.log(d); // 输出5
// if语句和for语句中用var定义的变量可以在外面访问到，可见，if语句和for语句属于块作用域，不属于函数作用域。

(1)let

let定义的变量，只能在块作用域里访问，不能跨块访问，也不能跨函数访问。
例. {
let a = 1;
}
console.log(a) //会报错 ReferenceError a is not defined

let不存在变量声明提升
例. {
console.log(a); //会报错，实际上它提升了，但是ES6规定了在let声明变量前不能使用该变量。
let a = 1;
}

一旦let在作用域中声明变量，就将统治这个作用域
例. var a = 1;
{
console.log(a); //会报错，块作用域里已有let声明a，形成了暂时性死区
let a = 1;
}

使用let解决for循环中i的闭包问题
var a = []
for(let i = 0;i < 10;i++) {
a[i] = function () {
console.log(i);
}
}
a1; //输出1，用let声明变量i，则当前i只在本轮循环内有效，每次循环块内的i其实都是一个新的变量

(2)const

const用来定义常量，使用时必须初始化(即必须赋值)，只能在块作用域里访问，而且不能修改。
例. const a = 1;
a = 2; //报错，不允许改

const a; //报错，必需马上赋值初始化
a = 1;

const a = 1;
let b = 1;
const a = 2;  //报错，不可重复声明
let b = 2;  //报错

当const指向变量的地址, 只要变量名所引用的地址不变就不会报错。
例. const a = {};
a.name = ‘nigel’; //不会报错
a = {}; //报错，因为改变了a的引用地址

解构赋值

ES6允许按照一定模式，从数组和对象中提取，对（var let const）变量进行赋值，这被称为解构。
例. var [a, b, c] = [1, 2, 3];
console.log(a, b, c); //输出 1 2 3

本质是上述匹配属于一种模式匹配 也就是只要等号两边的模式相同，左边的变量就会被赋予对应的值。
例. let [head,…tail] = [1,2,3,4,5];
let [a,b,…z] = [‘a’];

console.log(head,tail);  //输出 1 [2,3,4,5]
console.log(a,b,z);     //输出 a undefined []

(1)解构失败和不完全解构
解构失败是undefined -> 等号右边的值少了
不完全解构 -> 等号右边的值多了

(2)发生报错的情况
当等号右边不是可遍历结构时会报错
例. let [foo] = 1;
let [foo] = false;
let [foo] = NaN;
let [foo] = undefined;
let [foo] = null;
let [foo] = {};

(3)默认赋值
只有等号右边的值严格等于 undefined 的时候才生效

例. var [a, b = 1] = [1];
console.log(a, b); //输出 1 1
var [c, d = 1] = [1, undefined];
console.log(c, d); //输出 1 1 

var [c = 1] = [null];
console.log(c);    //输出 null

(4)惰性求值
只有在用到时，才会求值

例. function f () {
return 1;
}
var [a = f()] = [1];
var [b = f()] = [];
console.log(a, b) //输出 1 1

(5)默认赋值的变量赋值
先看等号右边有无和左边对应的值，没有的话看左侧从左到右解析

例. let [x1 = 1, y1 = x1] = [];
let [x2 = 1, y2 = x2] = [2];
let [x3 = 1, y3 = x3] = [1,2];
let [x4 = y4, y4 = 1] = []; //var 和 let 有区别
console.log(x1,y1); // 1 1
console.log(x2,y2); // 2 2
console.log(x3,y3); // 1 2
console.log(x4,y4); // 用let直接报错，若默认赋值y4给x4，可y4未定义; 用var 输出 undefined 1

(6)对象的解构赋值
对象的解构赋值是按照属性名称决定的

例. var {foo,bar} = {foo: ‘aaa’, bar: ‘bbb’};
console.log(foo, bar); //输出 aaa bbb
var {baz} = {foo: ‘aaa’, bar: ‘bbb’};
console.log(baz); //输出 undefined

应留意的地方：例. var x;
{x} = {x: 1}; //错误写法,js引擎会认为{x}是一个代码块，从而发生语法错误
({x} = {x: 1}); //应该用()包起来

当变量名与属性名不同时，可以通过以下方式赋值：
例. var {foo: baz} = {foo: 1}; //实际意图是为baz赋值，利用了foo属性名，因此等号两边的属性名要保持一致
console.log(foo); //会报错，因为foo不是变量名，只是一个属性名，相当于一种模式
console.log(baz); //输出 1

对象结构也可以自定默认值 默认生效条件和数组一样都是undefined
例. var {z = 3} = {};
console.log(z = 3); //输出 3
var {x, y = x} = {x:1};
console.log(x,y); //输出 1 1

(7)对象的解构赋值
字符串会转化成一个类数组的对象
例. var [a, b, c] = ‘hello’;
console.log(a, b, c); //输出 h e l

(8)类数组对象的解构赋值
类数组对象有length属性
例. let {length : len} = ‘hello’;
console.log(len); //输出 5 ，等号左边 length 属性对应’hello’的 length

(9)number和boolean的解构赋值
如果等号右边是数值或是布尔值，则会先转为对象
例. let {toString: a} = 123;
let {toString: b} = true;
console.log( a === Number.prototype.toString); //输出 true，123转化为number对象
console.log( b === Boolean.prototype.toString); //输出 true，true转化为boolean对象

let {toString: x } = undefined;
let {toString: y } = null;   //会报错，undefined 和 null不能转化为对象

(10)函数参数的解构赋值
函数add的参数表面上是一个数组，但在传入参数的那一刻，数组参数就被结构成变量 x 和 y
例. function add([x, y]) {
return x + y;
}
console.log(add([1,2])); //输出 3

函数参数的解构也可以使用默认值
例. function move({x = 0,y = 0} = {}) { //{}是指函数默认值，即没传参数时参数默认为{}
console.log([x,y]);
}
move({x:3,y:8}); //[3,8]
move({x:3}); //[3,0]
move({}); //[0,0] ，实参为{}，则对{}进行解构赋值，x y皆为默认值 0
move(); //[0,0] ，没有传参，函数默认实参为{}，对{}进行解构赋值

拓展例. function move({x, y} = { x: 0, y: 0 }) { //函数默认参数为{ x: 0, y: 0 }
console.log([x, y]);
}
move({x: 3, y: 8}); // [3, 8]
move({x: 3}); // [3, undefined]
move({}); // [undefined, undefined]
move(); // [0, 0]

(11)解构赋值的用途
1.交换变量
var x = 1;
var y = 2;
[x, y] = [y, x];

2.从函数返回多个值
function test () {
return [1,2,3];
}
var [a,b,c] = test();

3.定义函数的参数
function f({x, y, z}) {
console.log(x,y,z);
}
f({z: 1, x: 2, y: 3}); //输出 2 3 1

4.提交json数据
var jsonData= {
id: 42,
status: ‘ok’,
data:[1,2]
}
let {id, status, data:number} = jsonData;
console.log(id, status, number); //输出 42 ‘ok’ [1,2]

数组扩展

1.Array.form
该方法用于将两类对象转为真正的数组：类似数组的对象(所谓类似数组的对象，本质特征只有一点，即必须有length属性)
和 可遍历的对象(ES6新增的数据结构 Set和Map)

Array.from方法会将数组的空位，转为undefined，也就是说，这个方法不会忽略空位。
例. let arrayLike = {‘0’:’a’,’1’:’b’,’2’:’c’,length:3};
let arr1 = Array.prototype.slice.call(arrayLike);
let arr2 = Array.from(arrayLike);
console.log(arr1); //[“a”,”b”,”c”]
console.log(arr2); //[“a”,”b”,”c”]
console.log(Array.from({length:1})); //undefined

如果参数是一个真正的数组，Array.from会返回一个一模一样的新数组;
还可以接受第二个参数，作用类似于数组的map方法，用来对每个元素进行处理然后放入返回的数组之中。
例. var obj = {0:1,1:2,2:3,length:3};
var arr = Array.from(obj,x => x*x);
console.log(arr); //[1,4,9]

2.Array.of
该方法用于一组值，转换为数组。主要目的是弥补数组构造函数Array()的不足，因为参数个数的不同，会导致Array()的行为有差异。
例. console.log(Array.of(3, 11, 8)); //[3,11,8]
console.log(Array.of(3)); //[3]
console.log(Array.of(undefined)); //[undefined]
console.log(Array.of()); //[]

3.数组实例的copyWithin
在当前数组内部，将指定位置的成员复制到其他位置(会覆盖原有成员)，然后返回当前数组，该方法会修改当前数组。

arr.copyWithin(target, start, end)
target（必需）：从该位置开始替换数据。
start（可选） ：从该位置开始读取数据，默认为0。如果为负值，表示倒数。
end（可选） ：到该位置前停止读取数据(该位不读)，默认等于数组长度。如果为负值，表示倒数。

例. var arr = [1,2,3,4,5,6,7,8,9];
arr.copyWithin(0,3,6);
console.log(arr); //[4,5,6,4,5,6,7,8,9]

4.数组实例的find 和 findIndex
arr.find 用于找出第一个符合条件的数组成员。
它的参数是一个回调函数，所有数组成员依次执行该回调函数，直到找出第一个返回值为true的成员，
然后返回该成员。如果没有符合条件的成员，则返回undefined。
例. var arr = [1,5,10,15];
var num = arr.find(function(value,index,arr) {
return value > 9;
});
console.log(num); //10

arr,findIndex 用法与find方法非常类似。
返回第一个符合条件的数组成员的位置，如果所有成员都不符合条件，则返回-1。
例. var arr = [1,5,10,15];
var index = arr.findIndex(function (value, index, arr) {
return value > 9;
})
alert(index); //2

5.数组实例的fill
使用给定值，填充一个数组，该方法改变原数组。
例. var arr = [‘a’,’b’,’c’];
console.log(arr.fill(7)); //[7,7,7]

还可以接受第二个参数和第三个参数，用于指定填充的起始位置和结束位置。
例. console.log(new Array(5).fill(3,2,5)); //[empty,empty,3,3,3]

6.entries()、keys()、values()
用于遍历数组，可以用for…of循环进行遍历。
唯一的区别是keys()是对键名的遍历，values()是对键值的遍历，entries()是对键值对的遍历。
例. for (let index of [‘a’, ‘b’].keys()) {
console.log(index); //0 1
}
for (let elem of Object.values([‘a’, ‘b’])) { //values只能用Object调用
console.log(elem); // a b
}
for (let [index, elem] of [‘a’, ‘b’].entries()) {
console.log(index + ‘:’ + elem); //0:a 1:b
}

7.数组实例的includes
Array.prototype.includes方法返回一个boolean值，表示该数组是否包含给定的值，与字符串的includes方法类似
例. var a = [1,2,3];
var b = [1,2,3];
var c = [1,2,NaN];
console.log(a.includes(2)); //true
console.log(b.includes(4)); //false
console.log(c.includes(NaN)); //true
注：(1)includes方法不区分-0和+0
(2)include 和 indexOf 的区别是indexOf识别不了NaN 和 undefined ,都会返回-1

对象拓展

ES6允许直接写入变量和函数来作为对象的属性和方法，这样的书写更加简洁。

(1)ES6允许在对象之中，只写属性名，不写属性值。这时，属性值等于属性名所代表的变量。
例. var foo = ‘bar’;
var baz1 = {foo};
console.log(baz1); // {foo:’bar’}; 

function f1(x,y) {
    return {x,y};
}    
console.log(f1(1,2));  // {x:1, y:2}

var o1 = {
    method() {
        return 'hello!';
    }
}

(2)ES6允许字面量定义对象时，把表达式放在方括号内，用表达式作为对象的属性名。
例. let nigel = ‘boy’;
let obj = {

    [nigel]: 1
}
obj['a' + 'bc'] = 2;
console.log(obj);   //{boy: 1, abc: 2}

(3)Object.is
ES5比较两个值是否相等，只有两个运算符：相等运算符（==）和 严格相等运算符（===）。
它们都有缺点，前者会自动转换数据类型，后者的NaN不等于自身，以及+0等于-0。

ES6提出”Same-value equality”（同值相等）算法，Object.is就是部署这个算法的新方法。
它用来比较两个值是否严格相等，与严格比较运算符（===）的行为基本一致，是 === 的增强版。
例. console.log(+0 === -0); //true
console.log(NaN === NaN); //false
console.log(Object.is(+0, -0)); //false
console.log(Object.is(NaN,NaN));//true

(4)Object.assign
相当于对象的浅拷贝 Object.assign(target, …sources)
例. var target = {a:1};
var source1 = {b:2};
var source2 = {c:3};
Object.assign(target,source1,source2);
alert(target); //[object Object]
console.log(target); //{a:1, b:2, c:3}

注：对于这种嵌套的对象，一旦遇到同名属性，Object.assign的处理方法是替换，而不是添加。
例. var target = {a: { b: ‘c’, d: ‘e’ } };
var source = {a: { b: ‘hello’ } };
Object.assign(target, source);
console.log(target); //{a: {b: ‘hello’}}

Object.assign常用于为属性指定默认值
例. const DEFAULTS = {
logLevel: 0,
outputFormat: ‘html’
};
function processContent(options) {
options = Object.assign({}, DEFAULTS, options);
console.log(options);
// …
}

(5)proto属性
Object.setPrototypeOf(obj, newProto) 将obj对象的proto替换为newProto

Object.getPrototypeOf(obj) 获取obj对象的proto

(6)引用扩展运算符于对象
ES2017 将扩展运算符（…）用于取出参数对象的所有可遍历属性，拷贝到当前对象之中。
例. let z = { a: 3, b: 4 };
let n = { …z };
console.log(n); //{ a: 3, b: 4 }

函数拓展

(1)rest参数
ES6引入rest参数（形式为”…变量名”），用于获取函数的多余参数，这样就不需要使用arguments对象了。
例. function func(…values) {
var sum = 0;
for (var value in values) {
sum += values[value]; //加法器
}
return sum;
}

function func(array,...items) {
    items.forEach(function (item) {
        array.push(item);    //把多余的参数多放在array里
    })
}

注：rest参数之后不能再有其他参数（即只能是最后一个参数），否则会报错。

扩展运算符成就了链接数据的新写法
Es5 ： [1, 2, 3].concat(more)
Es6 ： [1,2, 3, …more]
例. var num = [1, 2, 4];
function add(x,y,z) {
return x + y + z;
}
add(…num); //返回7 ，…num-> 1, 2, 4 将一个数组转成最后分割的参数序列

(2)箭头函数
1.语义化更好：var x = v => v 等同于 var x = function (v) {return v;}

当箭头函数不需要参数或需要多个参数时，可使用一个圆括号代表参数部分
var f = () => 5 等同于 var f = function () {return 5;}
var sum = (sum1, sum2) => {return sum1 + sum2} 等同于 var sum = function (sum1, sum2) {return sum1 + sum2}

注：如果箭头函数的代码块部分多于一条语句，就要使用大括号讲将他们括起来，并且使用return语句返回

由于大括号被解释为代码块 所以如果箭头函数直接返回一个对象的话必须加上一个大括号
var fun = id => ({id:id, name:name})

例. function insert(value) {
return {into: function (array) {
return {after: function (afterValue) {
array.splice(array.indexOf(afterValue) + 1, 0, value);
return array;
}};
}};
}
可简化为：
var insert = (value) => ({into: (array) => ({after: (afterValue) => {
array.splice(array.indexOf(afterValue) + 1, 0, value);
return array;
}})});

2.this指向
需要注意的几点：
（1）箭头函数根本没有自己的this，导致内部的this就是外层代码块的this，继承的是父执行上下文里面的this，
而不是使用时所在的对象；
（2）没有自己的this，所以不可以当作构造函数，也不能使用new命令，否则会抛出一个错误；
（3）不可以使用arguments对象，该对象在函数体内不存在。如果要用，可以用Rest参数代替；

例. function foo() {
setTimeout(() => { //若是在setTimeout中调用普通函数，相当于自执行，在window下执行
console.log(‘id:’, this.id);
}, 100);
}
var id = 21; //window下的
foo.call({ id: 42 }); //输出 id:42，箭头函数的this指向foo执行的this，由于call调用foo执行的this指向{ id: 42 }

var a = {
    b: ()=>{
        console.log(this);
    }
}
a.b();   //输出 window，箭头函数的this继承自外部代码块的this，即a的this，但a是对象，不存在this(this只有在函数
          被调用，或者通过构造函数new Object()的形式才会有this)，而a的父执行上下文就是window。

(3)Class类
ES6提供了更接近传统语言的写法，引入了Class（类）这个概念，作为对象的模板。通过class关键字，可以定义类。
基本上，ES6的class可以看作只是一个语法糖，它的绝大部分功能，ES5都可以做到。
新的class写法只是让对象原型的写法更加清晰、更像面向对象编程的语法而已。

ES5：function Point (x,y) {
this.x = x;
this.y = y;
}
Point.prototype.toString = function () {
return ‘(‘ + this.x + ‘, ‘ + this.y + ‘)’;
}
var oP = new Point(1,2);

ES6: class Point { //前面不需要加上function这个保留字
constructor(x, y) { //相当于构造函数Point，类的数据类型就是函数，类本身就指向构造函数
this.x = x;
this.y = y;
}
toString() {
return ‘(‘ + this.x + ‘, ‘ + this.y + ‘)’;
}
}
var oP = new Point(1,2);

1.constructor方法
类的默认方法，通过new命令生成对象实例时，自动调用该方法。
一个类必须有constructor方法，如果没有显式定义，一个空的constructor方法会被默认添加。
constructor方法默认返回实例对象（即this），也可以指定返回另外一个对象。
例. class Foo{
constructor () {
return Object.create(null); //默认返回实例对象，但是更改后就是新的对象
}
}
console.log(new Foo() instanceof Foo); //false ，返回空对象后已与Foo脱离关系

2.prototype
类的所有方法其实都定义在类的prototype属性上面。
例. class Point {
constructor() {
this.x = 10;
alert(0);
}
toString() {
alert(‘1’);
}
toValue() {
alert(‘2’);
}
}
console.log(Point.prototype); //{constructor: ƒ, toString: ƒ, toValue: ƒ}

3.内部方法不可枚举
类的内部所有定义的方法，都是不可枚举的（enumerable）。
例. class Point {
constructor (x,y) {
this.x = x;
this.y = y;
}
toString() {
console.log(1);
}
}
var oP = new Point(1,2);
for (var name in oP) {
console.log(name); //无输出，无法枚举
}

4.类的实例共享一个原型对象
例. class Point {}
var p1 = new Point();
p1.proto.say = function () {console.log(‘Nigel’)};
var p2 = new Point();
p1.say(); //Nigel
p2.say(); //Nigel

5.类的name属性
例. class Point{}
console.log(Point.name); //Point

const MyClass = class Me { //从类的外部来看该类的名字是MyClass而不是Me，Me只在Class的内部代码可用，指代当前类。
    getClassName() {
        return Me.name;
    }
}   
let oM = new MyClass();
console.log(oM.getClassName());  //Me
Me.name    //会报错，Me在类外部不可用
console.log(MyClass.name);    //Me

注：Class不存在变量提升，这一点与ES5完全不同。

6.class的继承
ES5的继承，实质是先创造子类的实例对象this，然后再将父类的方法添加到this上面 Parent.apply(this)。
ES6的继承机制完全不同，实质是先创造父类的实例对象this（先调用super方法），然后再用子类的构造函数修改this。
例. class Parent {
constructor(x = 10,y = 15) { //函数默认赋值 x为10 y为15
this.x = x;
this.y = y;
}
toString() {
console.log(‘Parent toString’);
}
}

class Child extends Parent {
    constructor (x) {
        var a = super(x);  //子类必须在constructor方法中调用super方法，否则新建实例时会报错。
        console.log(a);      子类没有自己的this对象，而是继承父类的this对象，然后对其进行加工。
        this.age = 22;
        console.log(this.x, this.y);
    }
    toValue() {
        console.log('Child toString');
    }
}
var c = new Child(1);  //{age:22, x:1, y:15}  1 15
c.toValue();         //Child toString
c.toString();        //Parent toString
console.log(c.age);  //22

7.Class的取值函数 和 存值函数
prop属性有对应的存值函数和取值函数，因此赋值和读取行为都被自定义了。
例. class MyClass {
constructor() {
// …
}
get prop() {
return ‘getter’;
}
set prop(value) {
console.log(‘setter: ‘+value);
}
}
let inst = new MyClass();
inst.prop = 123; //setter: 123

8.Class的静态方法
如果在一个方法前，加上static关键字，就表示该方法不会被实例继承，而是直接通过类来调用，这就称为”静态方法”。
例. class Parent {
static classMethod() {
console.log(‘hello’);
}
}
Parent.classMethod(); // hello

var p = new Parent();
p.classMethod();  //报错，不存在该方法

class Bar extends Foo {}
Bar.classMethod();  // hello  父类的静态方法，可以被子类继承

静态方法也可以从super对象中调用
例. class Parent2 {
static classMethod() {
return ‘Parent2 staic’;
}
}
class Child2 extends Parent2 {
constructor () {
super();
}
static classMethod() {
console.log(super.classMethod() + ‘, too’); //子类调用父类的static方法只能在静态函数中调用
}
say() {
console.log(super.classMethod() + ‘, say’); //错误引用父类的静态方法
}
}
var oB2 = new Bar2();
oB2.say(); //报错
Bar2.classMethod(); //Parent2 staic, too

Set和Map数据结构

一.Set
ES6提供了新的数据结构 Set。它类似于数组，但是成员的值都是唯一的，没有重复的值。
Set本身是一个构造函数，用来生成 Set 数据结构。

Set 函数可以接受一个数组（或类似数组的对象）作为参数，用来初始化，返回类数组。
例. const set = new Set([1, 2, 3, 4, 4]);
console.log(set); //{0:1, 1:2, 2:3, 3:4, length：4}

去除数组重复成员的方法： […new Set(array)]

向Set加入值的时候，不会发生类型转换，比如5和”5”是两个不同的值。
Set内部判断两个值是否不同，使用的算法叫做”Same-value equality”，类似于精确相等运算符（===），
主要的区别是NaN等于自身，而精确相等运算符认为NaN不等于自身。

(1)Set实例的属性
Set.prototype.constructor：构造函数，默认就是Set函数。
Set.prototype.size：返回Set实例的成员总数。

(2)Set实例的方法
add(value,…)：添加一个或多个值，返回Set结构本身。
delete(value)： 删除某一个值，返回一个布尔值，表示删除是否成功。
has(value)： 返回一个布尔值，表示Set是否拥有该成员。
clear()： 清除所有成员，没有返回值。

(3)Array与Set的转换
Array.from方法可以将 Set 结构转为 Array结构。
const items = new Set([1, 2, 3, 4, 5]);
const array = Array.from(items);
另一种数组去重的方法

function dedupe(array) {
    return Array.from(new Set(array));
}
dedupe([1, 1, 2, 3]);

(4)Set实例的遍历方法
Set 结构的实例有四个遍历方法，可以用于遍历成员。

keys()： 返回键名的遍历器
values()： 返回键值的遍历器
entries()：返回键值对的遍历器
由于 Set 结构没有键名，只有键值（或者说键名和键值是同一个值），所以keys方法和values方法的行为完全一致。
例.
var set = new Set([1,22,4,4]);
for(var i of set.entries()){
console.log(i); //1 22 4
}

forEach()：使用回调函数遍历每个成员
Set结构的实例的forEach方法，用于对每个成员执行某种操作，没有返回值。

set.forEach(function (i) {
    console.log(i);
})

二.Map
Object 本质上是键值对的集合(Hash 结构)，但是传统上只能用字符串当作键。

ES6 提供了 Map 数据结构。它类似于对象，也是键值对的集合，但”键”不限于字符串，也可以是各种类型的值(包括对象)。

Object 结构提供了”字符串—值”的对应，Map结构提供了”值—值”的对应，是一种更完善的 Hash 结构实现。

构造函数 Map 接受一个数组作为参数，该数组的成员是一个个表示键值对的数组。

const map = new Map([
[‘name’, ‘nigel’],
[‘title’, ‘Author’]
]);

Map构造函数接受数组作为参数，实际上执行的是下面的算法：

const map = new Map();
items.forEach(
([key, value]) => map.set(key, value)
);

注：如果对同一个键多次赋值，后面的值将覆盖前面的值，如 map.set(1, ‘aaa’).set(1, ‘bbb’);

(1)Map实例的属性
map.size属性 返回 Map 结构的成员总数

(2)Map实例的方法
map.set(key, value) 设置键名key对应的键值为value，然后返回整个 Map 结构。
如果key已经有值，则键值会被更新，否则就新生成该键。
set方法返回的是当前的Map对象，因此可以采用链式写法。

get(key) 返回 key 对应的键值，如果找不到key，返回undefined。

has(key) 返回一个布尔值，表示某个键是否在当前 Map 对象之中。

delete(key) 删除某个键，成功返回true，删除失败返回false。

clear() 清除所有成员，没有返回值。

(3)Map实例的遍历方法
keys()： 返回键名的遍历器。
values()： 返回键值的遍历器。
entries()：返回所有成员的遍历器。
forEach()：遍历 Map 的所有成员。

Promise

JavaScript 的引擎建立在单线程事件轮询概念之上，单线程意味着一段时间内只能执行一段代码，
与 Java 和 C++ 这些允许多段代码同时执行的多线程语言形成了鲜明对比。

JavaScript 引擎在相同的时间内只能执行一段代码，所以引擎不需要追踪这些可能运行的代码，而是在它们准备好执行时将它们放置到任务队列。
当代码由 JavaScript 引擎执行完毕后，引擎通过 event loop 找到并执行队列中的下一个任务。
event loop 是 JavaScript 引擎内部的线程用来监控代码的执行情况和管理任务队列。
需要牢记的是既然它是个队列，那么任务就会由开始到最后的顺序依次执行。

(1)事件模型
当一个用户点击了一个按钮或按下一个键盘上的某个按键时，一个事件如 onclick 会被触发。
为了响应该交互，或许一个新的任务会被添加到任务队列中。这就是 JavaScript 异步编程中最基本的形式。
关于处理事件的代码只在事件发生后才会执行，此时相应的上下文（context）会出现。
比如:
let button = document.getElementById(“my-btn”);
button.onclick = function(event) {
console.log(“Clicked”);
};
在该段代码中，console.log(“Clicked”）在按钮被点击之前不会执行。
而在点击之后，赋值给 onclick 的代码会被添加到任务队列并在先前的所有任务完成之后执行。

(2)回调模式
在 Node.js 诞生后，它通过在编程中广泛使用回调模式来进一步发展异步编程模型。
回调模式和事件模型有些相似，异步的代码在之后的某一时刻才会执行，而它们的差异在于前者要调用的函数是参数。
不过当回调函数嵌套过多时，就会陷入了回调地狱（callback hell），这就需要Promise了。

(3)Promise的定义
所谓Promise，就是一个容器，里面保存着某个未来才会结束的事件（通常是一个异步操作）的结果。
从语法上说，Promise 是一个对象，从它可以获取异步操作的消息。

(4)Promise对象的特点
1.对象的状态不受外界影响。Promise对象代表一个异步操作，有三种状态：Pending（进行中）、Resolved（已完成，又称Fulfilled）
和 Rejected（已失败）。只有异步操作的结果，可以决定当前是哪一种状态，任何其他操作都无法改变这个状态。

2.一旦状态改变，就不会再变，任何时候都可以得到这个结果。Promise对象的状态改变，只有两种可能：从Pending 变为Resolved和
从Pending 变为 Rejected。只要这两种情况发生，状态就不会再变，会一直保持这个结果。如果改变已经发生了，你再对Promise对
象添加回调函数，也会立即得到这个结果。这与事件（Event）完全不同，事件的特点是，如果你错过了它，再去监听是得不到结果的。

(5)Promise的使用
Promise是一个构造函数，生成Promise对象

var promise = new Promise(function (resolve, reject){  //Promise以一个函数为参数，该函数的两个参数resolve reject
    if(/*异步操作成功*/){                                 是由JS引擎提供的两个函数
        resolve(value);
    }else{
        reject(err);
    }
});
//resolve函数将Promise对象的状态由'未完成'变成'成功'，并将异步操作成功后的结果作为参数传递出去；
  而reject函数是将状态由'未完成'变成'失败'，也将操作失败的错误信息作为参数传递出去。

promise.then(function (data) {  //promise状态为成功的回调函数，函数参数data就是上面的value
    //return 一个新的promise对象，且默认状态为resolve
}, function (err) {    //promise状态为失败的回调函数，函数参数err就是上面的err

}).then(...)   //这个then之前返回的是一个新的Promise对象，因此可以实现链式调用

then里的成功或失败回调函数，都可以返回了一个异步操作，即新的Promise对象，其状态决定下一个then调用哪个函数

var p1 = new Promise(function (resolve, reject) {
    setTimeout(() => reject(new Error('fail')), 3000)
})

var p2 = new Promise(function (resolve, reject) {
    setTimeout(() => resolve(p1), 1000)
})

p2.then(result => console.log(result)).catch(error => console.log(error))
//p1是一个Promise，3秒后状态变为rejected。p2的状态在1秒后变为resolve，resolve方法返回的是p1。
  由于p2用 resolve 返回 p1（另一个 Promise），导致p2自己的状态无效了，由p1的状态决定p2的状态。
  所以，后面的 then 语句和 catch 语句都变成针对 p1的状态。
    注：catch 方法是.then(null, rejection)的别名，用于指定发生错误时的回调函数，与前面的then
        指向同一个 Promise对象(p1)，由于 p1是 reject状态，而 then没有定义失败回调函数，所以直
        接触发catch。
  3秒后 p1变为rejected，导致触发catch方法指定的回调函数，输出 Error: fail。

Iterator

遍历器（Iterator）就是这样一种机制。它是一种接口，为各种不同的数据结构提供统一的访问机制。
任何数据结构只要部署Iterator接口，就可以完成遍历操作（即依次处理该数据结构的所有成员）。

一、Iterator的遍历过程
(1)创建一个指针对象，指向当前数据结构的起始位置。也就是说，遍历器对象本质上，就是一个指针对象。

(2)第一次调用指针对象的next方法，可以将指针指向数据结构的第一个成员。

(3)第二次调用指针对象的next方法，指针就指向数据结构的第二个成员。

(4)不断调用指针对象的next方法，直到它指向数据结构的结束位置。

注：每一次调用next方法，都会返回数据结构的当前成员的信息。具体来说，就是返回一个包含value和done两个属性的对象。
其中，value属性是当前成员的值，done属性是一个布尔值，表示遍历是否结束。

下面模拟next方法返回值

function makeIterator (arr) {
var index = 0;
return {
next () {
return index < arr.length ? {
value: arr[index++], done: false
} : {
value: undefined, done: true
}
}
}
}
var it = makeIterator([‘a’, ‘b’]);
it.next(); // { value: “a”, done: false }
it.next(); // { value: “b”, done: false }
it.next(); // { value: undefined, done: true }

二、Symbol.iterator
ES6规定，默认的Iterator接口部署在数据结构的Symbol.iterator属性，或者说，
一个数据结构只要具有Symbol.iterator属性，就可以认为是”可遍历的”。

var arr = [1, 2, 3];
var it = arrSymbol.iterator;
console.log(it.next()); //{ value: 1, done: false }

当使用for…of循环遍历某种数据结构时，该循环会自动去寻找Iterator接口。
在ES6中，有三类数据结构原生具备Iterator接口：数组、类数组对象、Set和Map结构。
也就是说，对象没有Symbol.iterator，也就无法使用for…of，对象只能用for…in遍历。

三、调用Iterator接口的场合
(1)对数组和Set结构进行解构赋值时，会默认调用Symbol.iterator方法。
(2)扩展运算符(…)也会调用默认的iterator接口。
(3)for…of
(4)Array.from()
(5)Map(), Set(), 比如new Map([[‘a’,1],[‘b’,2]])。

Generator

Generator 函数是 ES6 提供的一种异步编程解决方案，是一个状态机，封装了多个内部状态。
执行 Generator 函数会返回一个遍历器对象，依次遍历 Generator 函数内部的每一个状态。

function makeGenerator(a) { //键字与函数名之间有一个星号 ,就定义了Generator
var b = yield a + 1; //函数体内部使用yield语句，定义不同的内部状态
var c = yield b + 2;
return c;
}
var it = makeGenerator(2); //为第一行的yield后的a赋值
console.log(it.next()); //执行到第一行yield 输出 {value: 3, done: false}
console.log(it.next(4)); //为第一行yield的(a+1)赋值4，不赋值b是undefined，输出{value: 6, done: false}
console.log(it.next(77)); //为第二行yield后的(b+2)赋值77，输出 {value: 77, done: true}

一、遍历器对象的next方法的运行逻辑
(1)遇到yield语句，就暂停执行后面的操作，并将紧跟在yield后面的那个表达式的值，作为返回的对象的value属性值。

(2)下一次调用next方法时，再继续往下执行，直到遇到下一个yield语句。

(3)如果没有遇到新的yield语句，就一直运行到函数结束，直到return语句，并将return后面表达式的值，作为返回的对象的value属性值。

(4)如果该函数没有return语句，则返回的对象的value属性值为undefined。

二、for…of
for…of循环可以自动遍历Generator函数时生成的Iterator对象，且此时不再需要调用next方法。

三、异步操作的同步化表达
Generator函数的暂停执行的效果，意味着可以把异步操作写在yield语句里面，等到调用next方法时再往后执行。
这实际上等同于不需要写回调函数了，因为异步操作的后续操作可以放在yield语句下面，反正要等到调用next方
法时再执行。所以，Generator函数的一个重要实际意义就是用来处理异步操作，改写回调函数。
Ajax是典型的异步操作，通过Generator函数部署Ajax操作，可以用同步的方式表达。

function *myGenerator () {
var result = yield request(“url”);
var resp = JSON.parse(result);
console.log(resp.value);
}
function request(url) {
makeAjaxCall(url, function(response){
it.next(response);
});
}
var it = myGenerator();
it.next();

Module模块化

在 ES6 之前，社区制定了一些模块加载方案，最主要的有 CommonJS 和 AMD 两种。前者用于服务器，后者用于浏览器。
ES6 在语言标准的层面上，实现了模块功能，而且实现得相当简单，完全可以取代 CommonJS 和 AMD 规范，成为浏览器
和服务器通用的模块解决方案。

CommonJS 和 AMD 模块，都只能在运行时确定这些东西。比如，CommonJS 模块就是对象，输入时必须查找对象属性。

let { stat, exists, readFile } = require(‘fs’);

ES6 模块不是对象，而是通过export命令显式指定输出的代码，再通过import命令输入，称为静态加载。

import { stat, exists, readFile } from ‘fs’;

一、export命令
一个模块就是一个独立的文件。该文件内部的所有变量，外部无法获取。
如果你希望外部能够读取模块内部的某个变量，就必须使用export关键字输出该变量。

一个 JS 文件，里面使用export命令输出变量。
// profile.js
var firstName = ‘Michael’;
var lastName = ‘Jackson’;
var year = 1958;
export {firstName, lastName, year};

export命令除了输出变量，还可以输出函数或类（class）。
通常情况下，export输出的变量就是本来的名字，但是可以使用as关键字重命名。
function v1() { … }
function v2() { … }
export {
v1 as streamV1,
v2 as streamV2,
};

二、import命令
使用export命令定义了模块的对外接口以后，其他 JS 文件就可以通过import命令加载这个模块。

// main.js
import {firstName, lastName, year} from ‘./profile’;
function setName(element) {
element.textContent = firstName + ‘ ‘ + lastName;
}

如果想为输入的变量重新取一个名字，import命令要使用as关键字，将输入的变量重命名。
import { lastName as surname } from ‘./profile’;

注：import命令具有提升效果，会提升到整个模块的头部，首先执行。

还可以使用整体加载，即用星号（）指定一个对象，所有输出值都加载在这个对象上面。
import as circle from ‘./circle’;
console.log(‘圆面积：’ + circle.area(4));
console.log(‘圆周长：’ + circle.circumference(14));

三、export default 命令
为了给用户提供方便，让他们不用阅读文档就能加载模块，就要用到export default命令，为模块指定默认输出。

// export-default.js
export default function () {
console.log(‘foo’);
}
上面代码是一个模块文件export-default.js，它的默认输出是一个函数

其他模块加载该模块时，import命令可以为该匿名函数指定任意名字。
import customName from ‘./export-default’;
customName();
需要注意的是，这时import命令后面，不使用大括号。
