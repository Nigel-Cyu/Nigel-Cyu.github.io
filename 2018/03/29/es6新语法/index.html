es6新语法

Veröffentlicht am 2018-03-29

块级作用域

ES5里，JS中作用域有：全局作用域、函数作用域，没有块作用域的概念。ECMAScript 6(简称ES6)中新增了块级作用域。

块级作用域由 {} 包括，if语句和for语句里面的 {} 也属于块作用域。（严格模式下 function 变量属于块级作用域）

例如 {
var a = 1;
}
console.log(a); // 输出1，可见通过var定义的变量可以跨块作用域访问到。

(function A() {
    var b = 2;
    console.log(b); // 输出2
})();
console.log(b); // 报错，可见通过var定义的变量不能跨函数作用域访问到

if(true) {
    var c = 3;
}
console.log(c); // 3
for(var i = 0; i < 4; i ++) {
    var d = 5;
};
console.log(i); // 输出4   (循环结束i已经是4，所以此处i为4)
console.log(d); // 输出5
// if语句和for语句中用var定义的变量可以在外面访问到，可见，if语句和for语句属于块作用域，不属于函数作用域。

(1)let

let定义的变量，只能在块作用域里访问，不能跨块访问，也不能跨函数访问。
例. {
let a = 1;
}
console.log(a) //会报错 ReferenceError a is not defined

let不存在变量声明提升
例. {
console.log(a); //会报错，实际上它提升了，但是ES6规定了在let声明变量前不能使用该变量。
let a = 1;
}

一旦let在作用域中声明变量，就将统治这个作用域
例. var a = 1;
{
console.log(a); //会报错，块作用域里已有let声明a，形成了暂时性死区
let a = 1;
}

使用let解决for循环中i的闭包问题
var a = []
for(let i = 0;i < 10;i++) {
a[i] = function () {
console.log(i);
}
}
a1; //输出1，用let声明变量i，则当前i只在本轮循环内有效，每次循环块内的i其实都是一个新的变量

(2)const

const用来定义常量，使用时必须初始化(即必须赋值)，只能在块作用域里访问，而且不能修改。
例. const a = 1;
a = 2; //报错，不允许改

const a; //报错，必需马上赋值初始化
a = 1;

const a = 1;
let b = 1;
const a = 2;  //报错，不可重复声明
let b = 2;  //报错

当const指向变量的地址, 只要变量名所引用的地址不变就不会报错。
例. const a = {};
a.name = ‘nigel’; //不会报错
a = {}; //报错，因为改变了a的引用地址

解构赋值

ES6允许按照一定模式，从数组和对象中提取，对（var let const）变量进行赋值，这被称为解构。
例. var [a, b, c] = [1, 2, 3];
console.log(a, b, c); //输出 1 2 3

本质是上述匹配属于一种模式匹配 也就是只要等号两边的模式相同，左边的变量就会被赋予对应的值。
例. let [head,…tail] = [1,2,3,4,5];
let [a,b,…z] = [‘a’];

console.log(head,tail);  //输出 1 [2,3,4,5]
console.log(a,b,z);     //输出 a undefined []

(1)解构失败和不完全解构
解构失败是undefined -> 等号右边的值少了
不完全解构 -> 等号右边的值多了

(2)发生报错的情况
当等号右边不是可遍历结构时会报错
例. let [foo] = 1;
let [foo] = false;
let [foo] = NaN;
let [foo] = undefined;
let [foo] = null;
let [foo] = {};

(3)默认赋值
只有等号右边的值严格等于 undefined 的时候才生效

例. var [a, b = 1] = [1];
console.log(a, b); //输出 1 1
var [c, d = 1] = [1, undefined];
console.log(c, d); //输出 1 1 

var [c = 1] = [null];
console.log(c);    //输出 null

(4)惰性求值
只有在用到时，才会求值

例. function f () {
return 1;
}
var [a = f()] = [1];
var [b = f()] = [];
console.log(a, b) //输出 1 1

(5)默认赋值的变量赋值
先看等号右边有无和左边对应的值，没有的话看左侧从左到右解析

例. let [x1 = 1, y1 = x1] = [];
let [x2 = 1, y2 = x2] = [2];
let [x3 = 1, y3 = x3] = [1,2];
let [x4 = y4, y4 = 1] = []; //var 和 let 有区别
console.log(x1,y1); // 1 1
console.log(x2,y2); // 2 2
console.log(x3,y3); // 1 2
console.log(x4,y4); // 用let直接报错，若默认赋值y4给x4，可y4未定义; 用var 输出 undefined 1

(6)对象的解构赋值
对象的解构赋值是按照属性名称决定的

例. var {foo,bar} = {foo: ‘aaa’, bar: ‘bbb’};
console.log(foo, bar); //输出 aaa bbb
var {baz} = {foo: ‘aaa’, bar: ‘bbb’};
console.log(baz); //输出 undefined

应留意的地方：例. var x;
{x} = {x: 1}; //错误写法,js引擎会认为{x}是一个代码块，从而发生语法错误
({x} = {x: 1}); //应该用()包起来

当变量名与属性名不同时，可以通过以下方式赋值：
例. var {foo: baz} = {foo: 1}; //实际意图是为baz赋值，利用了foo属性名，因此等号两边的属性名要保持一致
console.log(foo); //会报错，因为foo不是变量名，只是一个属性名，相当于一种模式
console.log(baz); //输出 1

对象结构也可以自定默认值 默认生效条件和数组一样都是undefined
例. var {z = 3} = {};
console.log(z = 3); //输出 3
var {x, y = x} = {x:1};
console.log(x,y); //输出 1 1

(7)对象的解构赋值
字符串会转化成一个类数组的对象
例. var [a, b, c] = ‘hello’;
console.log(a, b, c); //输出 h e l

(8)类数组对象的解构赋值
类数组对象有length属性
例. let {length : len} = ‘hello’;
console.log(len); //输出 5 ，等号左边 length 属性对应’hello’的 length

(9)number和boolean的解构赋值
如果等号右边是数值或是布尔值，则会先转为对象
例. let {toString: a} = 123;
let {toString: b} = true;
console.log( a === Number.prototype.toString); //输出 true，123转化为number对象
console.log( b === Boolean.prototype.toString); //输出 true，true转化为boolean对象

let {toString: x } = undefined;
let {toString: y } = null;   //会报错，undefined 和 null不能转化为对象

(10)函数参数的解构赋值
函数add的参数表面上是一个数组，但在传入参数的那一刻，数组参数就被结构成变量 x 和 y
例. function add([x, y]) {
return x + y;
}
console.log(add([1,2])); //输出 3

函数参数的解构也可以使用默认值
例. function move({x = 0,y = 0} = {}) { //{}是指函数默认值，即没传参数时参数默认为{}
console.log([x,y]);
}
move({x:3,y:8}); //[3,8]
move({x:3}); //[3,0]
move({}); //[0,0] ，实参为{}，则对{}进行解构赋值，x y皆为默认值 0
move(); //[0,0] ，没有传参，函数默认实参为{}，对{}进行解构赋值

拓展例. function move({x, y} = { x: 0, y: 0 }) { //函数默认参数为{ x: 0, y: 0 }
console.log([x, y]);
}
move({x: 3, y: 8}); // [3, 8]
move({x: 3}); // [3, undefined]
move({}); // [undefined, undefined]
move(); // [0, 0]

(11)解构赋值的用途
1.交换变量
var x = 1;
var y = 2;
[x, y] = [y, x];

2.从函数返回多个值
function test () {
return [1,2,3];
}
var [a,b,c] = test();

3.定义函数的参数
function f({x, y, z}) {
console.log(x,y,z);
}
f({z: 1, x: 2, y: 3}); //输出 2 3 1

4.提交json数据
var jsonData= {
id: 42,
status: ‘ok’,
data:[1,2]
}
let {id, status, data:number} = jsonData;
console.log(id, status, number); //输出 42 ‘ok’ [1,2]

数组扩展

1.Array.form
该方法用于将两类对象转为真正的数组：类似数组的对象(所谓类似数组的对象，本质特征只有一点，即必须有length属性)
和 可遍历的对象(ES6新增的数据结构 Set和Map)

Array.from方法会将数组的空位，转为undefined，也就是说，这个方法不会忽略空位。
例. let arrayLike = {‘0’:’a’,’1’:’b’,’2’:’c’,length:3};
let arr1 = Array.prototype.slice.call(arrayLike);
let arr2 = Array.from(arrayLike);
console.log(arr1); //[“a”,”b”,”c”]
console.log(arr2); //[“a”,”b”,”c”]
console.log(Array.from({length:1})); //undefined

如果参数是一个真正的数组，Array.from会返回一个一模一样的新数组;
还可以接受第二个参数，作用类似于数组的map方法，用来对每个元素进行处理然后放入返回的数组之中。
例. var obj = {0:1,1:2,2:3,length:3};
var arr = Array.from(obj,x => x*x);
console.log(arr); //[1,4,9]

2.Array.of
该方法用于一组值，转换为数组。主要目的是弥补数组构造函数Array()的不足，因为参数个数的不同，会导致Array()的行为有差异。
例. console.log(Array.of(3, 11, 8)); //[3,11,8]
console.log(Array.of(3)); //[3]
console.log(Array.of(undefined)); //[undefined]
console.log(Array.of()); //[]

3.数组实例的copyWithin
在当前数组内部，将指定位置的成员复制到其他位置(会覆盖原有成员)，然后返回当前数组，该方法会修改当前数组。

arr.copyWithin(target, start, end)
target（必需）：从该位置开始替换数据。
start（可选） ：从该位置开始读取数据，默认为0。如果为负值，表示倒数。
end（可选） ：到该位置前停止读取数据(该位不读)，默认等于数组长度。如果为负值，表示倒数。

例. var arr = [1,2,3,4,5,6,7,8,9];
arr.copyWithin(0,3,6);
console.log(arr); //[4,5,6,4,5,6,7,8,9]

4.数组实例的find 和 findIndex
arr.find 用于找出第一个符合条件的数组成员。
它的参数是一个回调函数，所有数组成员依次执行该回调函数，直到找出第一个返回值为true的成员，
然后返回该成员。如果没有符合条件的成员，则返回undefined。
例. var arr = [1,5,10,15];
var num = arr.find(function(value,index,arr) {
return value > 9;
});
console.log(num); //10

arr,findIndex 用法与find方法非常类似。
返回第一个符合条件的数组成员的位置，如果所有成员都不符合条件，则返回-1。
例. var arr = [1,5,10,15];
var index = arr.findIndex(function (value, index, arr) {
return value > 9;
})
alert(index); //2

5.数组实例的fill
使用给定值，填充一个数组，该方法改变原数组。
例. var arr = [‘a’,’b’,’c’];
console.log(arr.fill(7)); //[7,7,7]

还可以接受第二个参数和第三个参数，用于指定填充的起始位置和结束位置。
例. console.log(new Array(5).fill(3,2,5)); //[empty,empty,3,3,3]

6.entries()、keys()、values()
用于遍历数组，可以用for…of循环进行遍历。
唯一的区别是keys()是对键名的遍历，values()是对键值的遍历，entries()是对键值对的遍历。
例. for (let index of [‘a’, ‘b’].keys()) {
console.log(index); //0 1
}
for (let elem of Object.values([‘a’, ‘b’])) { //values只能用Object调用
console.log(elem); // a b
}
for (let [index, elem] of [‘a’, ‘b’].entries()) {
console.log(index + ‘:’ + elem); //0:a 1:b
}

7.数组实例的includes
Array.prototype.includes方法返回一个boolean值，表示该数组是否包含给定的值，与字符串的includes方法类似
例. var a = [1,2,3];
var b = [1,2,3];
var c = [1,2,NaN];
console.log(a.includes(2)); //true
console.log(b.includes(4)); //false
console.log(c.includes(NaN)); //true
注：(1)includes方法不区分-0和+0
(2)include 和 indexOf 的区别是indexOf识别不了NaN 和 undefined ,都会返回-1

对象拓展

ES6允许直接写入变量和函数来作为对象的属性和方法，这样的书写更加简洁。

(1)ES6允许在对象之中，只写属性名，不写属性值。这时，属性值等于属性名所代表的变量。
例. var foo = ‘bar’;
var baz1 = {foo};
console.log(baz1); // {foo:’bar’}; 

function f1(x,y) {
    return {x,y};
}    
console.log(f1(1,2));  // {x:1, y:2}

var o1 = {
    method() {
        return 'hello!';
    }
}

(2)ES6允许字面量定义对象时，把表达式放在方括号内，用表达式作为对象的属性名。
例. let nigel = ‘boy’;
let obj = {

    [nigel]: 1
}
obj['a' + 'bc'] = 2;
console.log(obj);   //{boy: 1, abc: 2}

(3)Object.is
ES5比较两个值是否相等，只有两个运算符：相等运算符（==）和 严格相等运算符（===）。
它们都有缺点，前者会自动转换数据类型，后者的NaN不等于自身，以及+0等于-0。

ES6提出”Same-value equality”（同值相等）算法，Object.is就是部署这个算法的新方法。
它用来比较两个值是否严格相等，与严格比较运算符（===）的行为基本一致，是 === 的增强版。
例. console.log(+0 === -0); //true
console.log(NaN === NaN); //false
console.log(Object.is(+0, -0)); //false
console.log(Object.is(NaN,NaN));//true

(4)Object.assign
相当于对象的浅拷贝 Object.assign(target, …sources)
例. var target = {a:1};
var source1 = {b:2};
var source2 = {c:3};
Object.assign(target,source1,source2);
alert(target); //[object Object]
console.log(target); //{a:1, b:2, c:3}

注：对于这种嵌套的对象，一旦遇到同名属性，Object.assign的处理方法是替换，而不是添加。
例. var target = {a: { b: ‘c’, d: ‘e’ } };
var source = {a: { b: ‘hello’ } };
Object.assign(target, source);
console.log(target); //{a: {b: ‘hello’}}

Object.assign常用于为属性指定默认值
例. const DEFAULTS = {
logLevel: 0,
outputFormat: ‘html’
};
function processContent(options) {
options = Object.assign({}, DEFAULTS, options);
console.log(options);
// …
}

(5)proto属性
Object.setPrototypeOf(obj, newProto) 将obj对象的proto替换为newProto

Object.getPrototypeOf(obj) 获取obj对象的proto

(6)引用扩展运算符于对象
ES2017 将扩展运算符（…）用于取出参数对象的所有可遍历属性，拷贝到当前对象之中。
例. let z = { a: 3, b: 4 };
let n = { …z };
console.log(n); //{ a: 3, b: 4 }

函数拓展

(1)rest参数
ES6引入rest参数（形式为”…变量名”），用于获取函数的多余参数，这样就不需要使用arguments对象了。
例. function func(…values) {
var sum = 0;
for (var value in values) {
sum += values[value]; //加法器
}
return sum;
}

function func(array,...items) {
    items.forEach(function (item) {
        array.push(item);    //把多余的参数多放在array里
    })
}

注：rest参数之后不能再有其他参数（即只能是最后一个参数），否则会报错。

扩展运算符成就了链接数据的新写法
Es5 ： [1, 2, 3].concat(more)
Es6 ： [1,2, 3, …more]
例. var num = [1, 2, 4];
function add(x,y,z) {
return x + y + z;
}
add(…num); //返回7 ，…num-> 1, 2, 4 将一个数组转成最后分割的参数序列

(2)箭头函数
1.语义化更好：var x = v => v 等同于 var x = function (v) {return v;}

当箭头函数不需要参数或需要多个参数时，可使用一个圆括号代表参数部分
var f = () => 5 等同于 var f = function () {return 5;}
var sum = (sum1, sum2) => {return sum1 + sum2} 等同于 var sum = function (sum1, sum2) {return sum1 + sum2}

注：如果箭头函数的代码块部分多于一条语句，就要使用大括号讲将他们括起来，并且使用return语句返回

由于大括号被解释为代码块 所以如果箭头函数直接返回一个对象的话必须加上一个大括号
var fun = id => ({id:id, name:name})

例. function insert(value) {
return {into: function (array) {
return {after: function (afterValue) {
array.splice(array.indexOf(afterValue) + 1, 0, value);
return array;
}};
}};
}
可简化为：
var insert = (value) => ({into: (array) => ({after: (afterValue) => {
array.splice(array.indexOf(afterValue) + 1, 0, value);
return array;
}})});

2.this指向
需要注意的几点：
（1）箭头函数根本没有自己的this，导致内部的this就是外层代码块的this，继承的是父执行上下文里面的this，
而不是使用时所在的对象；
（2）没有自己的this，所以不可以当作构造函数，也不能使用new命令，否则会抛出一个错误；
（3）不可以使用arguments对象，该对象在函数体内不存在。如果要用，可以用Rest参数代替；

例. function foo() {
setTimeout(() => { //若是在setTimeout中调用普通函数，相当于自执行，在window下执行
console.log(‘id:’, this.id);
}, 100);
}
var id = 21; //window下的
foo.call({ id: 42 }); //输出 id:42，箭头函数的this指向foo执行的this，由于call调用foo执行的this指向{ id: 42 }

var a = {
    b: ()=>{
        console.log(this);
    }
}
a.b();   //输出 window，箭头函数的this继承自外部代码块的this，即a的this，但a是对象，不存在this(this只有在函数
          被调用，或者通过构造函数new Object()的形式才会有this)，而a的父执行上下文就是window。

(3)Class类
ES6提供了更接近传统语言的写法，引入了Class（类）这个概念，作为对象的模板。通过class关键字，可以定义类。
基本上，ES6的class可以看作只是一个语法糖，它的绝大部分功能，ES5都可以做到。
新的class写法只是让对象原型的写法更加清晰、更像面向对象编程的语法而已。

ES5：function Point (x,y) {
this.x = x;
this.y = y;
}
Point.prototype.toString = function () {
return ‘(‘ + this.x + ‘, ‘ + this.y + ‘)’;
}
var oP = new Point(1,2);

ES6: class Point { //前面不需要加上function这个保留字
constructor(x, y) { //相当于构造函数Point，类的数据类型就是函数，类本身就指向构造函数
this.x = x;
this.y = y;
}
toString() {
return ‘(‘ + this.x + ‘, ‘ + this.y + ‘)’;
}
}
var oP = new Point(1,2);

1.constructor方法
类的默认方法，通过new命令生成对象实例时，自动调用该方法。
一个类必须有constructor方法，如果没有显式定义，一个空的constructor方法会被默认添加。
constructor方法默认返回实例对象（即this），也可以指定返回另外一个对象。
例. class Foo{
constructor () {
return Object.create(null); //默认返回实例对象，但是更改后就是新的对象
}
}
console.log(new Foo() instanceof Foo); //false ，返回空对象后已与Foo脱离关系

2.prototype
类的所有方法其实都定义在类的prototype属性上面。
例. class Point {
constructor() {
this.x = 10;
alert(0);
}
toString() {
alert(‘1’);
}
toValue() {
alert(‘2’);
}
}
console.log(Point.prototype); //{constructor: ƒ, toString: ƒ, toValue: ƒ}

3.内部方法不可枚举
类的内部所有定义的方法，都是不可枚举的（enumerable）。
例. class Point {
constructor (x,y) {
this.x = x;
this.y = y;
}
toString() {
console.log(1);
}
}
var oP = new Point(1,2);
for (var name in oP) {
console.log(name); //无输出，无法枚举
}

4.类的实例共享一个原型对象
例. class Point {}
var p1 = new Point();
p1.proto.say = function () {console.log(‘Nigel’)};
var p2 = new Point();
p1.say(); //Nigel
p2.say(); //Nigel

5.类的name属性
例. class Point{}
console.log(Point.name); //Point

const MyClass = class Me { //从类的外部来看该类的名字是MyClass而不是Me，Me只在Class的内部代码可用，指代当前类。
    getClassName() {
        return Me.name;
    }
}   
let oM = new MyClass();
console.log(oM.getClassName());  //Me
Me.name    //会报错，Me在类外部不可用
console.log(MyClass.name);    //Me

注：Class不存在变量提升，这一点与ES5完全不同。

6.class的继承
ES5的继承，实质是先创造子类的实例对象this，然后再将父类的方法添加到this上面 Parent.apply(this)。
ES6的继承机制完全不同，实质是先创造父类的实例对象this（先调用super方法），然后再用子类的构造函数修改this。
例. class Parent {
constructor(x = 10,y = 15) { //函数默认赋值 x为10 y为15
this.x = x;
this.y = y;
}
toString() {
console.log(‘Parent toString’);
}
}

class Child extends Parent {
    constructor (x) {
        var a = super(x);  //子类必须在constructor方法中调用super方法，否则新建实例时会报错。
        console.log(a);      子类没有自己的this对象，而是继承父类的this对象，然后对其进行加工。
        this.age = 22;
        console.log(this.x, this.y);
    }
    toValue() {
        console.log('Child toString');
    }
}
var c = new Child(1);  //{age:22, x:1, y:15}  1 15
c.toValue();         //Child toString
c.toString();        //Parent toString
console.log(c.age);  //22

7.Class的取值函数 和 存值函数
prop属性有对应的存值函数和取值函数，因此赋值和读取行为都被自定义了。
例. class MyClass {
constructor() {
// …
}
get prop() {
return ‘getter’;
}
set prop(value) {
console.log(‘setter: ‘+value);
}
}
let inst = new MyClass();
inst.prop = 123; //setter: 123

8.Class的静态方法
如果在一个方法前，加上static关键字，就表示该方法不会被实例继承，而是直接通过类来调用，这就称为”静态方法”。
例. class Parent {
static classMethod() {
console.log(‘hello’);
}
}
Parent.classMethod(); // hello

var p = new Parent();
p.classMethod();  //报错，不存在该方法

class Bar extends Foo {}
Bar.classMethod();  // hello  父类的静态方法，可以被子类继承

静态方法也可以从super对象中调用
例. class Parent2 {
static classMethod() {
return ‘Parent2 staic’;
}
}
class Child2 extends Parent2 {
constructor () {
super();
}
static classMethod() {
console.log(super.classMethod() + ‘, too’); //子类调用父类的static方法只能在静态函数中调用
}
say() {
console.log(super.classMethod() + ‘, say’); //错误引用父类的静态方法
}
}
var oB2 = new Bar2();
oB2.say(); //报错
Bar2.classMethod(); //Parent2 staic,too
